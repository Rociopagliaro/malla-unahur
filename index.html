function iconoParaMateria(nombre) { if (nombre.toLowerCase().includes("matemática")) return ''; if (nombre.toLowerCase().includes("lógica")) return ''; if (nombre.toLowerCase().includes("computadoras") || nombre.toLowerCase().includes("arquitectura")) return ''; if (nombre.toLowerCase().includes("programación")) return ''; if (nombre.toLowerCase().includes("estructuras")) return ''; if (nombre.toLowerCase().includes("web") || nombre.toLowerCase().includes("marcado")) return ''; if (nombre.toLowerCase().includes("interfaces")) return ''; if (nombre.toLowerCase().includes("base") || nombre.toLowerCase().includes("persistencia")) return ''; if (nombre.toLowerCase().includes("ingeniería")) return ''; if (nombre.toLowerCase().includes("inglés")) return ''; if (nombre.toLowerCase().includes("electiva")) return ''; if (nombre.toLowerCase().includes("materia unahur")) return ''; return ''; }
